def make_move
  @game.draw
  puts "Where would you like to move?"
  pos = gets.chomp.to_i
  until @game.valid_move?(pos)
    puts "Sorry, that is not a valid move. Please try again."
    pos = gets.chomp.to_i
  end
  @game.mark(pos, @symbol.to_s)
end
